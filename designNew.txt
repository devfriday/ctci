Minimum implementation design
Java
public interface AiAdvisorService {
    AiConfigReview reviewConfigChange(ConfigChangeContext context);
    String generateMrDescription(ConfigChangeContext context, RiskReport riskReport);
    String generateReleaseNotes(MergedMrContext context);
}
Java
@Service
public class LangChain4jAiAdvisorService implements AiAdvisorService {

    private final ChatLanguageModel model;

    public LangChain4jAiAdvisorService(ChatLanguageModel model) {
        this.model = model;
    }

    @Override
    public AiConfigReview reviewConfigChange(ConfigChangeContext ctx) {
        String prompt = PromptTemplates.configReview(ctx);
        String response = model.chat(prompt);
        return parseJson(response);
    }
}
